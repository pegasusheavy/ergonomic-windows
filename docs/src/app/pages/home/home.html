Quick Example

use ergonomic_windows::prelude::*;

fn main() -> Result<()> {
    // String conversions are automatic
    let wide = WideString::new("Hello, Windows! ðŸªŸ");

    // Spawn a process with the builder pattern
    let process = Command::new("notepad.exe")
        .arg("file.txt")
        .spawn()?;

    // Read from the registry
    let version = registry::get_string(
        RootKey::LOCAL_MACHINE,
        "SOFTWARE\\Microsoft\\Windows NT\\CurrentVersion",
        "ProductName"
    )?;

    println!("Running on: {}", version);
    Ok(())
}
