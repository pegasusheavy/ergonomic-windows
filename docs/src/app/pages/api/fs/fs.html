File System

Windows-specific file system utilities and operations.

use ergonomic_windows::fs::*;

get_file_attributes

pub fn get_file_attributes(path: &Path) -> Result<FileAttributes>

Gets the Windows file attributes for a path. Returns a struct with methods to check individual attribute flags.

Example

use ergonomic_windows::fs::get_file_attributes;
use std::path::Path;

let attrs = get_file_attributes(Path::new("C:\\Windows"))?;

if attrs.is_directory() {
    println!("It's a directory");
}

if attrs.is_hidden() {
    println!("It's hidden");
}
