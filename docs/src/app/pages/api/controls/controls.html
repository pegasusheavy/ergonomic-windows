Win32 Common Controls

Safe wrappers for Windows common controls including buttons, edit boxes, list boxes, combo boxes, and progress bars.

Overview

The controls module provides ergonomic wrappers around Win32 common controls. These are the standard UI elements used in traditional Windows desktop applications.

Note: Controls require a parent window. Use the window module to create a parent window first.

Initialization

use ergonomic_windows::controls::init_common_controls;

// Initialize common controls (called automatically by control constructors)
init_common_controls()?;

Button

Create push buttons, checkboxes, radio buttons, and group boxes. 

use ergonomic_windows::controls::{Button, ButtonStyle, Control};

// Create a push button
let button = Button::new(
    parent_hwnd,
    "Click Me",
    10, 10,      // x, y position
    100, 30,     // width, height
    1001,        // control ID
    ButtonStyle::Push,
)?;

// Create a checkbox
let checkbox = Button::new(
    parent_hwnd,
    "Enable Feature",
    10, 50,
    150, 24,
    1002,
    ButtonStyle::AutoCheckbox,
)?;

// Check/uncheck programmatically
Button::set_checked(&checkbox, true);
let is_checked = Button::is_checked(&checkbox);

Edit (Text Box)

Create single-line and multi-line text input controls. 

use ergonomic_windows::controls::{Edit, EditStyle, TextAlign};

// Create a simple text box
let edit = Edit::new(
    parent_hwnd,
    "Default text",
    10, 100,
    200, 24,
    1003,
    EditStyle::default(),
)?;

// Create a password field
let password = Edit::new(
    parent_hwnd,
    "",
    10, 130,
    200, 24,
    1004,
    EditStyle {
        password: true,
        ..Default::default()
    },
)?;

// Create a multiline text area
let textarea = Edit::new(
    parent_hwnd,
    "",
    10, 160,
    300, 100,
    1005,
    EditStyle {
        multiline: true,
        auto_vscroll: true,
        ..Default::default()
    },
)?;

// Control operations
Edit::set_limit(&edit, 100);           // Max 100 characters
Edit::select_all(&edit);               // Select all text
Edit::replace_selection(&edit, "new"); // Replace selection
let (start, end) = Edit::selection(&edit);

Progress Bar

use ergonomic_windows::controls::{ProgressBar, ProgressStyle};

// Standard progress bar
let progress = ProgressBar::new(
    parent_hwnd,
    10, 200,
    300, 20,
    1007,
    ProgressStyle::Standard,
)?;

// Configure range and position
ProgressBar::set_range(&progress, 0, 100);
ProgressBar::set_pos(&progress, 50);

// Step-based progress
ProgressBar::set_step(&progress, 10);
ProgressBar::step(&progress); // Advance by step amount

// Marquee (indeterminate) progress bar
let marquee = ProgressBar::new(
    parent_hwnd,
    10, 230,
    300, 20,
    1008,
    ProgressStyle::Marquee,
)?;
ProgressBar::set_marquee(&marquee, true, 30); // Enable with 30ms interval

List Box

use ergonomic_windows::controls::ListBox;

// Single-select list box
let listbox = ListBox::new(
    parent_hwnd,
    10, 260,
    200, 120,
    1009,
    false, // multi_select
)?;

// Add items
ListBox::add_string(&listbox, "Item 1");
ListBox::add_string(&listbox, "Item 2");
ListBox::add_string(&listbox, "Item 3");
ListBox::insert_string(&listbox, 1, "Inserted at 1");

// Selection
ListBox::set_selected_index(&listbox, 0);
let selected = ListBox::selected_index(&listbox);

// Other operations
let count = ListBox::count(&listbox);
ListBox::delete_string(&listbox, 0);
ListBox::clear(&listbox);

Combo Box

use ergonomic_windows::controls::ComboBox;

// Dropdown list (read-only)
let combo = ComboBox::new(
    parent_hwnd,
    10, 400,
    200, 200, // Height includes dropdown
    1010,
    true, // dropdown (read-only)
)?;

// Editable dropdown
let editable_combo = ComboBox::new(
    parent_hwnd,
    10, 430,
    200, 200,
    1011,
    false, // editable dropdown
)?;

// Add items and manage selection
ComboBox::add_string(&combo, "Option 1");
ComboBox::add_string(&combo, "Option 2");
ComboBox::set_selected_index(&combo, 0);
let selected = ComboBox::selected_index(&combo);

Control Base Class

All controls share common functionality through the Control struct: 

// Show/hide
control.show();
control.hide();

// Enable/disable
control.enable();
control.disable();

// Get/set text
let text = control.text();
control.set_text("New text");

// User data (for associating custom data)
control.set_user_data(my_data as isize);
let data = control.user_data();

// Get raw handle (for interop)
let hwnd = control.hwnd();
